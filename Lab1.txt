    public static double squareRoot(double x) {
// creation and assignment of variables
// -------------------------
        double cotaSup, cotaInf = 0, puntoM = 0, root = 0, previous = 0;
        boolean end = false;
        cotaSup = (x > 1) ? x : 1;

        if (x != 0 && x != 1) {
            while (!end) {
                puntoM = (cotaSup + cotaInf) / 2;
                if ((puntoM * puntoM) > x) {
                    cotaSup = puntoM;
                } else if ((puntoM * puntoM) < x) {
                    cotaInf = puntoM;
                } else if ((puntoM * puntoM) == x) {
                    root = puntoM;
                }
                if (puntoM == previous) {
                    root = puntoM;
                    end = true;
                }
                if (!(Math.abs((x - (puntoM * puntoM))) > 0.00000000000001)) {
                    root = puntoM;
                    end = true;
                }
                previous = puntoM;
            }
        } else {
            if (x == 0) {
                return 0;
            } else {
                return 1;
            }
        }
        return root;
    }

    public static double factorial(double num1) {

        int result = 1;

        String binario = "";
        if (num1 > 0) {
            while (num1 > 0) {
                if (num1 % 2 == 0) {
                    binario = "0" + binario;
                } else {
                    binario = "1" + binario;
                }
                num1 = (int) num1 / 2;
            }
        } else if (num1 == 0) {
            binario = "0";
        } else {
            binario = "No se pudo convertir el numero. Ingrese solo números positivos";
        }

        return binario;
    }
